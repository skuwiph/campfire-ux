Notes

Html:

<app-cf-typeahead
    [placeholderTextDefault]="'Default text'"
    [placeholderTextActive]="'Enter search terms'"
    [isSearching]="isTypeaheadSearching"
    [hasResults]="typeaheadHasResults"
    (lookup)="doTypeaheadLookup($event)">
    <div results>
        <div *ngIf="typeaheadHasResults; else noData">
            <app-cf-table [data]="typeaheadSearchTable" [sortable]="true"
            (onSelect)="selectedStandardRow($event)"></app-cf-table>
        </div>
        <ng-template #noData>No data returned</ng-template>
    </div>
</app-cf-typeahead>
